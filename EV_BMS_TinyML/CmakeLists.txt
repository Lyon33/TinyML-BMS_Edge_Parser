cmake_minimum_required(VERSION 3.14)
project(EV_BMS_TinyML VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# 输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# 查找线程库
find_package(Threads REQUIRED)

# 头文件路径
include_directories(
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/third_party/json)

# 源文件
file(GLOB SOURCES 
    "${PROJECT_SOURCE_DIR}/src/*.cpp"
)

# 生成可执行文件
add_executable(bms_TinyML_parser ${SOURCES})

# 链接库（后续需要可以加 pthread、rt 等）
target_link_libraries(bms_TinyML_parser PRIVATE Threads::Threads)

# 编译选项
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
    target_compile_options(bms_TinyML_parser PRIVATE -Wall -Wextra -O2)
endif()

# 安装（可选）
install(TARGETS bms_TinyML_parser DESTINATION bin)

message(STATUS "✅ TinyML_BMS_Parser 项目配置完成！")
