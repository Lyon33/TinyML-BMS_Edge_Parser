cmake_minimum_required(VERSION 3.16)

# ==================== 彻底禁用系统 GTest ====================
set(CMAKE_DISABLE_FIND_PACKAGE_GTest ON CACHE BOOL "" FORCE)
set(CMAKE_DISABLE_FIND_PACKAGE_GMock ON CACHE BOOL "" FORCE)

project(EV_BMS_Z03 VERSION 1.2 LANGUAGES CXX)

# ==================== 全局设置 ====================
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
endif()

option(ENABLE_WARNINGS "Enable strict compiler warnings" ON)
option(ENABLE_TESTS "Build unit tests" ON)

# ==================== 依赖 ====================
find_package(Threads REQUIRED)

# ==================== GoogleTest（完全隔离） ====================
include(FetchContent)

# ✅ 强制不使用系统 gtest
set(CMAKE_USE_SYSTEM_GTEST OFF CACHE BOOL "" FORCE)

# ✅ 防止 Homebrew gtest 被捡到
set(GTEST_INCLUDE_DIRS "" CACHE STRING "" FORCE)
set(GTEST_LIBRARIES "" CACHE STRING "" FORCE)

set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
set(BUILD_GMOCK ON CACHE BOOL "" FORCE)
set(INSTALL_GTEST OFF CACHE BOOL "" FORCE)

FetchContent_Declare(
    googletest
    URL https://github.com/google/googletest/archive/refs/tags/v1.14.0.zip
    DOWNLOAD_EXTRACT_TIMESTAMP TRUE

)

FetchContent_MakeAvailable(googletest)

# ==================== nlohmann/json ====================
add_library(json_interface INTERFACE)
target_include_directories(json_interface INTERFACE
    ${CMAKE_SOURCE_DIR}/third_party/json
)

# ==================== BMS Core ====================
file(GLOB_RECURSE BMS_CORE_SOURCES
    "${CMAKE_SOURCE_DIR}/src/*.cpp"
)

list(FILTER BMS_CORE_SOURCES EXCLUDE REGEX ".*main\\.cpp$")

add_library(bms_core STATIC ${BMS_CORE_SOURCES})

target_include_directories(bms_core PUBLIC
    ${CMAKE_SOURCE_DIR}/include
)

target_link_libraries(bms_core
    Threads::Threads
    json_interface
)

# ==================== App ====================
add_executable(bms_parser src/main.cpp)
target_link_libraries(bms_parser PRIVATE bms_core)

# ==================== Warnings ====================
if(ENABLE_WARNINGS)
    target_compile_options(bms_core PRIVATE
        $<$<CXX_COMPILER_ID:GNU>:-Wall -Wextra -Werror>
        $<$<CXX_COMPILER_ID:Clang>:-Wall -Wextra -Werror>
    )
endif()

# ==================== Tests ====================
if(ENABLE_TESTS)
    enable_testing()

    file(GLOB TEST_SOURCES "${CMAKE_SOURCE_DIR}/test/*.cpp")

    add_executable(bms_tests ${TEST_SOURCES})

    target_link_libraries(bms_tests PRIVATE
        bms_core
        gtest
        gtest_main
        Threads::Threads
        json_interface
    )

    target_include_directories(bms_tests PRIVATE
        ${CMAKE_SOURCE_DIR}/include
    )

    add_test(NAME BMSTests COMMAND bms_tests)
    message(STATUS "✅ GoogleTest 已集成！运行测试: ./bin/bms_tests")
endif()

# ==================== Install ====================
install(TARGETS bms_parser DESTINATION bin)
install(FILES ${CMAKE_SOURCE_DIR}/config/bms_protocol.json DESTINATION config)

message(STATUS "🚀 EV BMS Z03 项目配置完成！版本 ${PROJECT_VERSION}")
